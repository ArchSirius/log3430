package tests;

	private ArrayList<Object> setA;

	@Before
	public void setUp() {
		// Create 4 sets

		setB = new ArrayList<Object>();
		setA.add(1);
		setB.add(2);

	* Cas de test 1
	* Sequence
	* 	setAt
	* Oracle : S0,S2

	@Test

			list.setAt(setA, 0);
			fail("Expected IndexOutOfBoundsException");
		}
		catch (final IndexOutOfBoundsException e) {
			assert true;
		}

		}
